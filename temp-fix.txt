  const availableRoles = ['all', ...Array.from(new Set(
    contacts.flatMap(contact => {
      const roleFields = ['Role', 'Position', 'Title', 'Job Title']
      return roleFields.map(field => {
        const value = contact.properties[field]
        if (value) {
          const valueStr = Array.isArray(value) ? value.join(' ') : String(value)
          // Extract key role types
          if (valueStr.toLowerCase().includes('head coach')) return 'head coach'
          if (valueStr.toLowerCase().includes('assistant coach')) return 'assistant coach'
          if (valueStr.toLowerCase().includes('athletic director')) return 'athletic director'
          if (valueStr.toLowerCase().includes('administrator')) return 'administrator'
          if (valueStr.toLowerCase().includes('coordinator')) return 'coordinator'
          if (valueStr.toLowerCase().includes('director')) return 'director'
          if (valueStr.toLowerCase().includes('commissioner')) return 'commissioner'
        }
        return null
      }).filter(Boolean)
    })
  ))]
